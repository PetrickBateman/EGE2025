from itertools import product, permutations

def f(x, w, z, y):
    return (not(x <= z)) or (y == w) or y # переписываем функцию из условия


for a1, a2, a3, a4, a5, a6, a7 in product([0, 1], repeat=7): # относительно количества пустых графов создаем переменные
    table = (
        (1, 0, a1, a2),
        (a3, 1, 0, a4),
        (0, a5, a6, a7)
    )
    if len(table) == len(set(table)): # проверяем на одинаковые строки
        for p in permutations('xzwy'):
            u = [f(**dict(zip(p, t))) for t in table] == [0, 0 ,0] # сверяем значение переменных с вертикалями
            if u:
                print(*p)
